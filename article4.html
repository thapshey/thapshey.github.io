བོད་ཁམས་ཕྱོགས་ཀྱི་རྫ་སྟོད་མི་མང་གིས་ཁོང་ཚོའི་ས་ཁུལ་གྱི་གཏེར་ཁའི་བཟོ་གྲྭ་དང་རུལ་སུངས་ཅན་གྱི་ས་གནས་སྲིད་གཞུང་གཉིས་ཀྱི་བྱེད་ལས་ལ་ངོ་རྒོལ་བྱས་ནས་རྒྱལ་ཁ་ཆེན་པོ་བླངས་བ་རེད། ཁོང་ཚོས་ཁོར་ཡུག་ལ་གནོད་པའི་གཏེར་ཁ་བསྔོགས་འདོན་གྱི་ལས་གཞི་ཁག་མཚམས་འཇོག་བྱེད་དུ་བཅུག་པ་མ་ཟད། པེ་ཅིང་སྲིད་གཞུང་གིས་ཀྱང་རྫ་སྟོད་ནི་ཁོར་ཡུག་སྲུང་སྐྱོབ་ཀྱི་ས་ཁུལ་ཡིིན་པར་བསྒྲགས།


འདི་ནས་རྫ་སྟོད་ཀྱི་གནད་དོན་དེ་ཀློགས།

ནམ་རྒྱུན་་བོད་མི་རྣམས་ཀྱིས་སོ་སོའི་ས་ཁུལ་གྱི་གནས་རི་རྩ་ཆེན་ཁག་སྲུང་སྐྱོབ་ཀྱི་ལས་འགུལ་སྤེལ་བ་ལམ་སློང་ཡོང་དཀའ་བ་ཡོད་པ་རེད། དེས་ན། རྫ་སྟོད་མི་མང་གིས་ལས་འགུལ་མ་འདྲ་བ་ཅི་ཞིག་སྤེལ་བ་རེད་དམ། ཁོང་ཚོར་ཇི་ལྟར་རྒྱལ་ཁ་འཐོབ་པ་རེད་དམ།

རྫ་སྟོད་ཀྱི་ལས་འགུལ་དེའི་ཁྱད་ཆོས་མ་འདྲ་བ་འགའ་ཤས་ནི། འཆར་གཞི་ལྡན་པ། ཐབས་ཇུས་ཡོད་པ། གནད་དོན་ལ་ཁ་གཏད་བྱེད་མཁས་པ་བཅས་རེད།


ཐོག་མར་རྫ་སྟོད་ཀྱི་ལས་འགུལ་བཀོད་སྒྲིག་པ་རྣམས་ཀྱིས་གཟབ་ནན་ངང་འཆར་གཞི་བཏིངས་པ་རེད། དེ་ཡང་། ཁོང་ཚོའི་དམིགས་ཡུལ་རྣམས་དངོས་ཡོད་དང་འཚམས་པ་མ་ཟད་དམིགས་ཡུལ་གྱི་ཁ་ཕྱོགས་ཀྱང་གསལ་བོ་རེད། ཁོང་ཚོས་སྲིད་འཛིན་ཞིས་ཅིན་ཕིན་གྱིས་གནས་སྐབས་ཀྱི་དཔལ་འབྱོར་ཡར་རྒྱས་ཀྱི་སླད་དུ་ཁོར་ཡུག་སྲུང་སྐྱོབ་ལ་གཏོར་བཤིག་གཏོང་མི་ཆོག་ཟེར་བའི་གསུང་བཤད་དེ་དང་རྫ་སྟོད་དེ་ཁོར་ཡུག་སྲུང་སྐྱོབ་ཀྱི་ས་ཁུལ་ཡིན་པའི་དངོས་ཡོད་གཉིས་ཐབས་ཇུས་ཀྱི་དཀྱིལ་སྙིང་ལ་བཟུང་ནས་ལས་འགུལ་སྤེལ་བ་རེད།


ཁོང་ཚོས་གོང་གི་དངོས་དོན་འདི་གཉིས་བེད་སྤྱོད་བཏང་ནས་རྫ་སྟོད་ཀྱི་གཏེར་ཁ་བསྔོགས་འདོན་གྱི་གནད་དོན་དེ་གླེང་བ་མ་ཟད། དེ་ཡང་ཆབ་སྲིད་ཀྱི་རྩོད་གཞི་སོགས་རྒྱང་དུ་དོར་ཅིང་ཁོར་ཡུག་སྲུང་སྐྱོབ་དང་ལྐོག་ཟ་རུལ་སུངས་ཀྱི་གནད་དོན་གཙོ་བོར་བཟུང་བ་རེད། ས་གནས་མང་ཚོག
