མེ་ལྕེ་རྒོད་པོ། བོད་མིའི་ལྤགས་རིགས་མེར་བསྲེགས་ཀྱི་ལས་འགུལ།

ཀརྨས་པགས་རིགས་མེར་བསྲེགས་ལས་འགུལ་སྐོར་གྱི་སྒྲུང་བཤད་པ། (སྒྲུང་འདི་ནས་ཀློགས།)


སྟོབས་མེད་ཀྱི་སྟོབས་ཤུགས་ཏེ། མང་ཚོགས་མཉམ་ཞུགས་ཀྱི་ལས་འགུལ།

སྤྱི་ལོ་༢༠༠༦ལོར་༧རྒྱལ་བ་རིན་པོ་ཆེས་བོད་མི་རྣམས་ལ་རྩ་ཆེའི་རི་དྭགས་རྣམས་ཀྱི་ལྤགས་རིགས་གྱོན་མཚམས་འཇོག་དགོས་བའི་འབོད་སྐུལ་གནང་བའི་རྗེས་སུ། བོད་ཁམས་ཡོངས་སུ་ལྤགས་རིགས་མེར་བསྲེགས་རྒྱ་ཆེར་བྱས་པ་མ་ཟད་། བོད་མིའི་ཆིག་སྒྲིལ་གྱི་བྱ་སྤྱོད་དེ་སྲིད་གཞུང་གིས་ཀྱང་འགོག་ཐུབ་ཡོད་མ་རེད། ལས་འགུལ་དེ་ལ་མཉམ་ཞུགས་བྱེད་མཁན་རྣམས་ཀྱིས་གནད་དོན་དེ་ཆབ་སྲིད་ཀྱི་ངོ་བོར་མི་འགྱུར་བའི་ཆེད་དུ།


ལྤགས་རིགས་མེར་བསྲེགས་ནི་ཕྱོགས་གཅིག་ནས་ནང་པའི་འཚེ་མེད་ཞི་བའི་ལྟ་གྲུབ་དང་མཐུན་པ་དང་ཕྱོགས་གཉིས་ནས་རྒྱལ་ཁབ་ཀྱི་ཁོར་ཡུག་སྲུང་སྐྱོབ་ཀྱི་ཁྲིམས་དང་མཐུན་པ་ཡིན་ཚུལ་ཤུགས་ཆེར་གླེང།


བོད་མི་རྣམས་ཀྱིས་གྱོན་གོས་ཀྱི་ཚབ་ལ་་ལྤགས་རིགས་གྱོན་པའི་ལོ་རྒྱུས་ནི་་ཧ་ཅང་རིང་པོ་ཡོད་ན་ཡང་། ལྤགས་རིགས་དང་རྒྱན་ཆའི་རིགས་ངོམ་སྲོལ་བྱེད་པའི་སྲོལ་དེ་ཕྱི་ལོ་༡༩༩༠ལོ་རབས་ནས་བྱུང་བ་ཞིག་ཡིན་ལ་དེ་ཡང་གཙོ་བོ་སྲིད་གཞུང་གི་སྐུལ་མ་ལ་བརྟེན་ནས་བྱུང་། གང་ལྟར། ཕྱི་ལོ་༢༠༠༦ལོའི་ལྤགས་རིགས་མེར་བསྲེགས་ཀྱི་ལས་འགུལ་དེས་རྒྱ་ནག་སྲིད་གཞུང་ཡང་སོམ་ཉིའི་ཀློང་ལ་ཚུད་པ་རེད། གང་ཡིན་ཞེ་ན། རྒྱ་ནག་ཕྱོགས་ནས་བོད་མི་རྣམས་ཀྱིས་༧རྒྱལ་བ་རིན་པོ་ཆེའི་བཀའ་ལ་སྲུང་བརྩི་བྱེད་བཞིན་པ་དེར་ལྟོས་ནས་བསྡད་ན་ཡང་མ་འགྲིག ལྤགས་རིགས་མ་གྱོན་ན་མི་འགྲིག་ཟེར་ནས་ཉེས་པ་བཏང་ན་དེ་ཁྲིམས་འགལ་ཆགས་ཀྱི་ཡོད་ཙང་ཁོང་ཚོ་ཅི་བྱ་གཏོལ་མེད་དུ་གྱུར་བ་རེད།


མཐའ་མ་དེར། རྒྱ་ནག་གཞུང་གིས་ལྤགས་རིགས་མེར་བསྲེགས་ཀྱི་ལས་འགུལ་ལ་ཞུགས་མཁན་གྱི་བོད་མི་འགའ་ཤས་བཀག་ཉར་བྱས་པ་མ་ཟད་ལྤགས་རིགས་མེར་བསྲེགས་ཀྱི་ལས་འགུལ་ཁ་ཤས་བཀག་པ་རེད། རྗེས་སུ་ཁོང་ཚོས་བོད་མི་རྣམས་ལ་ལྤགས་རིགས་གྱོན་དགོས་བའི་འབོད་སྐུལ་བྱས་པ་མ་ཟད་ས་ཁུལ་ཁ་ཤས་སུ་དུས་སྟོན་གྱི་སྐབས་སུ་་ལྤགས་རིགས་གྱོན་དགོས་བའི་བཙན་བཀའ་ཡང་བཏང་འདུག འོན་ཀྱང་ལྤགས་རིགས་མེར་བསྲེགས་ཀྱི་ལས་འགུལ་དེའི་མཇུག་འབྲས་དེ་བོད་མི་ཚོའི་སྟངས་འཛིན་གྱི་འོག་ན་ཡོད་པ་མ་གཏོགས་སྲིད་གཞུང་དང་བཟོ་ལས་ཁང་གི་བདག་པོ་སོགས་ཀྱི་ལག་ན་ཡོད་མ་རེད།


ལྤགས་རིགས་མེར་བསྲེགས་ཀྱི་ལས་འགུལ་འདི་ལྟ་བུ་ཐོག་མར་བོད་མི་སྒེར་བ་དང་གྲོང་གསེབ་སྒེར་བ་ནས་འགོ་ཚུགས་ཤིང་འདི་ག་ལེར་ཁྱབ་རྒྱ་ཇེ་ཆེར་སོང་ནས་བོད་མི་རྣམས་ལ་སྤྱི་མཐུན་གྱི་ནུས་ཤུགས་ཤིག་བསྐྲུན་པ་མ་ཟད་། བོད་མི་རྣམས་ཀྱི་རི་སྐྱེས་སྲོག་ཆགས་སྲུང་སྐྱོབ་ཀྱི་ཆོད་སེམས་དང་༧རྒྱལ་བ་རིན་པོ་ཆེར་བཟུང་བའི་དད་གུས་ཀྱང་གསལ་སྟོན་བྱས། ལྷག་པར་དུ། ལྤགས་རིགས་མེར་བསྲེགས་འདི་ཡིས་ཆོལ་ཁ་སོ་སོར་ཡོད་པའི་བོད་མི་རྣམས་ཀྱི་ཐུན་མོང་གི་རིན་ཐང་དང་བོད་མིའི་ཆིག་སྒྲིལ་དང་སྟོབས་ཤུགས་སོགས་ཀྱང་མངོན་དུ་བཅུག


ཁམས་ལི་ཐང་ནས་ཡིན་པའི་ལྤགས་རིགས་མེར་བསྲེགས་ལས་འགུལ་བ་ཞིག་གིས་སོ་སོའི་མྱོང་ཚོར་བརྗོད་དོན།

༧རྒྱལ་བ་རིན་པོ་ཆེ་ཡིས་ལྤགས་རིགས་མཚམས་འཇོག་གི་སྐོར་ལ་བཀའ་སློབ་ནན་ཏན་གནང་བ་དེས་ང་ཚོས་རི་སྐྱེས་སྲོག་ཆགས་ཀྱི་སྐོར་ལ་བསམ་བློ་གཏོང་སྟངས་ལ་ཤུགས་རྐྱེན་ཆེན་པོ་ཐེབས་པ་མ་ཟད་། ཁོང་གི་བཀའ་སློབ་ང་ཚོས་ལག་ལེན་བསྟར་ཐུབ་པ་དེས་ང་ཚོས་ཁོང་ལ་བརྩི་བཀུར་དང་ང་ཚོ་ནམ་ཡང་ཁོང་གི་ཐུགས་ཁུར་འོག་ཏུ་ཡོད་པ་དེ་སྟོན། ལྷག་པར་དུ། ང་ཚོའི་བྱེད་སྟངས་དེ་ནི་རྒྱ་ནག་ཁྲིམས་དང་རྩ་འགལ་མིན་པ་མ་ཟད་ང་ཚོས་ཁོང་ཚོའི་ཁྲིམས་ཀྱང་བེད་སྤྱོད་གཏོང་ཐུབ་ཡོད་པ་རེད།
